# DSL AST Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the hand-written DSL parser, generator, and schema renderer with a single schema-driven AST system where annotated zod schemas are the sole source of truth for parsing, rendering, completions, and editor interactions.

**Architecture:** A `dsl()` helper annotates zod schemas with DSL rendering/parsing hints via a WeakMap. A shared schema walker produces an AST from either a model (rendering) or token stream (parsing) using a strategy pattern. The AST replaces the flat span array — all editor interactions (click, hover, completions, decorations) become tree queries against the AST.

**Tech Stack:** TypeScript, Zod, React, CodeMirror 6, Vitest

**Spec:** `docs/superpowers/specs/2026-03-24-dsl-ast-design.md`

// src/dsl/dslMeta.ts
import type { z } from 'zod';

export interface PositionalHint {
  keys: string[];
  format?: 'dimension' | 'spaced' | 'joined' | 'arrow' | 'quoted' | 'tuples';
  separator?: string;
  suffix?: string;
  keyword?: string;           // intermediate keyword (e.g., 'at' in chapter "name" at 3.5)
  fallbackToKwarg?: boolean;  // when subset of keys present, emit as kwargs
  transform?: 'double';       // value transformation (e.g., radius→diameter for ellipse)
}

export interface DslHints {
  keyword?: string;
  positional?: PositionalHint[];
  kwargs?: string[];
  flags?: string[];
  sigil?: { key: string; prefix: string };
  children?: Record<string, 'block' | 'inline'>;
  record?: {
    key: string;
    entryHints: DslHints;
  };
  variants?: Array<{
    when: string;
    hints: DslHints;
  }>;
  // Node-line specific (only on NodeSchema):
  nodeId?: string;
  geometry?: string[];
  inlineProps?: string[];
  blockProps?: string[];
  inlineLayoutHints?: string[];
}

const dslRegistry = new WeakMap<z.ZodType, DslHints>();

export function dsl<T extends z.ZodType>(schema: T, hints: DslHints): T {
  dslRegistry.set(schema, hints);
  return schema;
}

export function getDsl(schema: z.ZodType): DslHints | undefined {
  return dslRegistry.get(schema);
}
```

// src/dsl/astTypes.ts
import type { z } from 'zod';

export type DslRole =
  | 'keyword' | 'value' | 'kwarg-key' | 'kwarg-value'
  | 'flag' | 'sigil' | 'separator' | 'compound'
  | 'document' | 'section';

export interface AstNode {
  schema?: z.ZodType;
  schemaPath: string;
  modelPath: string;
  from: number;
  to: number;
  value?: unknown;
  children: AstNode[];
  parent?: AstNode;
  dslRole: DslRole;
}

export function createAstNode(init: Partial<AstNode> & Pick<AstNode, 'dslRole' | 'from' | 'to' | 'schemaPath' | 'modelPath'>): AstNode {
  return {
    children: [],
    ...init,
  };
}

/** Find the deepest AST node containing the given position. */
export function nodeAt(root: AstNode, pos: number): AstNode | null {
  if (pos < root.from || pos >= root.to) return null;
  for (const child of root.children) {
    const found = nodeAt(child, pos);
    if (found) return found;
  }
  return root;
}

/** Walk up from a node to find the nearest compound ancestor (or self). */
export function findCompound(node: AstNode): AstNode | null {
  let current: AstNode | undefined = node;
  while (current) {
    if (current.dslRole === 'compound') return current;
    current = current.parent;
  }
  return null;
}

/** Flatten all leaf nodes (non-compound, non-document, non-section) sorted by position. */
export function flattenLeaves(root: AstNode): AstNode[] {
  const leaves: AstNode[] = [];
  const walk = (node: AstNode) => {
    if (node.children.length === 0 && node.dslRole !== 'document' && node.dslRole !== 'section') {
      leaves.push(node);
    }
    for (const child of node.children) walk(child);
  };
  walk(root);
  leaves.sort((a, b) => a.from - b.from);
  return leaves;
}
```

- [ ] **Step 2: Implement the walk strategy interface**

```typescript
// src/dsl/astWalker.ts
import type { z } from 'zod';
import type { AstNode, DslRole } from './astTypes';
import type { DslHints, PositionalHint } from './dslMeta';
import { getDsl } from './dslMeta';

/**
 * Walk a schema's DSL hints, calling the strategy for each element.
 * The strategy determines whether we're emitting (model→text) or consuming (text→model).
 *
 * Walk order: keyword → positional → sigil → kwargs → flags → children → record
 *
 * For schemas with variants, selects the matching variant based on the strategy's
 * variant detection (value inspection for emit, token lookahead for consume).
 */
export function walkSchema(
  schema: z.ZodType,
  hints: DslHints,
  strategy: WalkStrategy,
): void {
  // Resolve variants if present
  const effectiveHints = strategy.resolveVariant?.(hints) ?? hints;

  // Keyword
  if (effectiveHints.keyword) {
    strategy.handleKeyword(effectiveHints.keyword);
  }

  // Positional
  if (effectiveHints.positional) {
    for (const pos of effectiveHints.positional) {
      strategy.handlePositional(pos);
    }
  }

  // Sigil
  if (effectiveHints.sigil) {
    strategy.handleSigil(effectiveHints.sigil);
  }

  // Kwargs
  if (effectiveHints.kwargs) {
    for (const key of effectiveHints.kwargs) {
      strategy.handleKwarg(key);
    }
  }

  // Flags
  if (effectiveHints.flags) {
    for (const key of effectiveHints.flags) {
      strategy.handleFlag(key);
    }
  }

  // Children (block sub-schemas)
  if (effectiveHints.children) {
    for (const [key, mode] of Object.entries(effectiveHints.children)) {
      strategy.handleChildren(key, mode);
    }
  }

  // Record (dynamic-keyed maps)
  if (effectiveHints.record) {
    strategy.handleRecord(effectiveHints.record);
  }
}

export interface WalkStrategy {
  handleKeyword(keyword: string): void;
  handlePositional(hint: PositionalHint): void;
  handleSigil(sigil: { key: string; prefix: string }): void;
  handleKwarg(key: string): void;
  handleFlag(key: string): void;
  handleChildren(key: string, mode: 'block' | 'inline'): void;
  handleRecord(record: { key: string; entryHints: DslHints }): void;
  resolveVariant?(hints: DslHints): DslHints;
}
```

- [ ] **Step 2: Implement the consume strategy and buildAstFromText**

The `ConsumeStrategy` implements `WalkStrategy` for the text→model direction:
1. Takes a token stream from the existing tokenizer (`src/dsl/tokenizer.ts`)
2. Consumes tokens matching expected schema structure
3. Builds AST nodes with positions from consumed tokens
4. Extracts model values

`buildAstFromText(text)` orchestrates parsing:
1. Tokenizes text using existing `tokenize()`
2. Creates document AST node
3. Detects sections by top-level keywords (name, description, background, viewport, images, style, animate) and node definitions
4. For each section, walks the appropriate schema (NodeSchema for nodes, AnimConfigSchema for animate, etc.)
5. Extracts FormatHints by detecting inline vs block format per node
6. Returns `{ ast, model, formatHints }`

**Token consumption** uses the existing token types from `src/dsl/types.ts`. The parser peeks and consumes tokens matching the expected hint:
- `keyword` → expect `identifier` token matching the keyword string
- `dimension` → expect `dimensions` token (already parsed by tokenizer as `140x80`)
- `spaced` → expect N consecutive `number` tokens
- `joined` → expect values separated by separator tokens
- `arrow` → expect values separated by `arrow` tokens
- `kwarg` → expect `identifier` + `equals` + value token
- `flag` → expect `identifier` matching flag name
- `quoted` → expect `string` token
- `sigil` → expect `atSign` + `identifier`

**Union dispatch** during parsing: peek at next token, check if it matches any union member's keyword. If yes, use that member. If no keyword match, try structural detection (string token → string branch, number → number, etc.).

**Error recovery**: on unexpected token, skip to next newline and continue. Record error on the AST node.

**Key reference file:** `src/dsl/parser.ts` — the current parser. Port the section detection, indentation handling, and error recovery patterns. The token consumption logic replaces the hand-written recursive descent with schema-driven walks.

// src/dsl/astDecorations.ts
import { Decoration, type DecorationSet, EditorView } from '@codemirror/view';
import { StateField, StateEffect } from '@codemirror/state';
import type { AstNode } from './astTypes';
import { flattenLeaves, nodeAt } from './astTypes';

export const setAst = StateEffect.define<AstNode>();

export const astField = StateField.define<AstNode | null>({
  create: () => null,
  update(ast, tr) {
    for (const e of tr.effects) {
      if (e.is(setAst)) return e.value;
    }
    return ast;
  },
});

export const astDecorations = StateField.define<DecorationSet>({
  create: () => Decoration.none,
  update(decos, tr) {
    if (!tr.effects.some(e => e.is(setAst))) return decos;
    const ast = tr.state.field(astField);
    if (!ast) return Decoration.none;
    const leaves = flattenLeaves(ast);
    const marks = leaves
      .filter(n => n.from < n.to) // skip empty ranges
      .map(n => Decoration.mark({
        attributes: {
          'data-schema-path': n.schemaPath,
          'data-model-path': n.modelPath,
          'data-dsl-role': n.dslRole,
        },
      }).range(n.from, n.to));
    marks.sort((a, b) => a.from - b.from);
    return Decoration.set(marks);
  },
  provide: f => EditorView.decorations.from(f),
});

export function astExtension() {
  return [astField, astDecorations];
}

/** Convenience: find AST node at position from editor state */
export function astNodeAtPos(ast: AstNode | null, pos: number): AstNode | null {
  if (!ast) return null;
  return nodeAt(ast, pos);
}
